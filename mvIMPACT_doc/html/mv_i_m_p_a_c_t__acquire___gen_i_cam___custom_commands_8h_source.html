241  template<typename _Ty>
242  _Ty hostToNet16( _Ty value )
243  {
244  switch( O32HostByteOrderEvaluator_.get32BitHostByteOrder() )
245  {
246  case O32_LITTLE_ENDIAN:
247  return ( ( value & 0xff00 ) >> 8 ) | ( ( value & 0x00ff ) << 8 );
248  case O32_BIG_ENDIAN:
249  break;
250  }
251  return value;
252  }
253  template<typename _Ty>
254  _Ty hostToNet32( _Ty value )
255  {
256  switch( O32HostByteOrderEvaluator_.get32BitHostByteOrder() )
257  {
258  case O32_LITTLE_ENDIAN:
259  return ( ( value & 0xff000000 ) >> 24 ) | ( ( value & 0x00ff0000 ) >> 8 ) | ( ( value & 0x0000ff00 ) << 8 ) | ( ( value & 0x000000ff ) << 24 );
260  case O32_BIG_ENDIAN:
261  break;
262  }
263  return value;
264  }

299  {

302  {

323  catch( const ImpactAcquireException& e )

689  int queueSequencerSetValueModification(
691  const int64_type sequencerSet,
693  const TSequencerSetParameter parameter,
695  const double value )
696  {
697  const int result = queueCustomCommandPreCondition( sizeof( CustomCommandUpdateSequencerSetF ) );
698  prepareCustomCommandUpdateSequencerSet( commandBuffer_.get() + currentCommandBufferOffset_, ccUpdateSequencerSet, sizeof( CustomCommandUpdateSequencerSetF ) - sizeof( CustomCommandHeader ), sequencerSet, parameter );
699  CustomCommandUpdateSequencerSetF* pPacket = ( CustomCommandUpdateSequencerSetF* )( commandBuffer_.get() + currentCommandBufferOffset_ );
700  const size_t byteCount = sizeof( double ) < 8 ? sizeof( double ) : 8;
701  switch( O32HostByteOrderEvaluator_.get32BitHostByteOrder() )
702  {
703  case O32_LITTLE_ENDIAN:
704  {
705  for( size_t i = byteCount ; i > 0; i-- )
706  {
707  pPacket->data[i - 1] = ( reinterpret_cast<const char*>( &value ) )[byteCount - i];
708  }
709  }
710  break;
711  case O32_BIG_ENDIAN:
712  memcpy( pPacket->data, &value, byteCount );
713  break;
714  }
715  currentCommandBufferOffset_ += sizeof( CustomCommandUpdateSequencerSetF );
716  return result;
717  }

739  return result;
