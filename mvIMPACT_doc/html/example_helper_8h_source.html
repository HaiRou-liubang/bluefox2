348 template<class _Elem, class _Traits, class _Ax>
349 int match( const std::basic_string<_Elem, _Traits, _Ax>& searchString, const std::basic_string<_Elem, _Traits, _Ax>& candidate, _Elem wildcard )
350 //-----------------------------------------------------------------------------
351 {
352  typename std::basic_string<_Elem, _Traits, _Ax>::size_type searchLength = searchString.length();
353  // determine search length
354  if( candidate.length() < searchString.length() )
355  {
356  if( candidate.empty() )
357  {
358  return -1;
359  }
360 
361  if( candidate[candidate.length() - 1] != wildcard )
362  {
363  return -1;
364  }
365  searchLength = candidate.length() - 1;
366  }
367  // search
368  for( typename std::basic_string<_Elem, _Traits, _Ax>::size_type i = 0; i < searchLength; i++ )
369  {
370  if( ( candidate[i] != searchString[i] ) && ( candidate[i] != wildcard ) )
371  {
372  return -1;
373  }
374  }
375  return 0;
376 }
