157  std::streamsize write(
159  const char* pBuf,
161  int64_type offs,
163  int64_type len,
165  const char* pFileName )
166  {
167  m_ptrFileSelector.writeS( pFileName );
168  m_ptrFileOperationSelector.writeS( "Write" );
169 
170  const std::streamsize maxWriteLen( static_cast<std::streamsize>( m_ptrFileAccessBuffer.binaryDataBufferMaxSize() ) );
171  std::streamsize bytesWritten = 0;
172  while ( bytesWritten < len )
173  {
174  // copy streamdata to xchange buffer
175  const int64_type remain( len - bytesWritten );
176  const int64_type writeSize( remain <= maxWriteLen ? remain : maxWriteLen );
177 
178  // offset
179  if( offs + bytesWritten > m_ptrFileAccessOffset.getMaxValue() )
180  {
181  // we cannot move the outbuffer any further
182  break;
183  }
184 
185  m_ptrFileAccessOffset.write( offs + bytesWritten );

187  // set the buffer
188  const std::string bufS( pBuf + bytesWritten, static_cast<unsigned int>( Align( writeSize, 4 ) ) );
189  m_ptrFileAccessBuffer.writeBinary( bufS );
190  m_ptrFileOperationExecute.call();
191  bytesWritten += static_cast<std::streamsize>( m_ptrFileOperationResult.read() );
192  if( m_ptrFileOperationStatus.readS() != "Success" )
193  {
194  // no more space on device
195  break;
196  }

204  std::streamsize read(
206  char* pBuf,
208  int64_type offs,
210  std::streamsize len,
212  const char* pFileName )
213  {
214  m_ptrFileSelector.writeS( pFileName );
215  m_ptrFileOperationSelector.writeS( "Read" );
216  const std::streamsize maxReadLen = static_cast<std::streamsize>( m_ptrFileAccessBuffer.binaryDataBufferMaxSize() );
217  std::streamsize bytesRead = 0;
218  while( bytesRead < len )
219  {
220  // copy xchange buffer to streamdata
221  const std::streamsize readSize = std::min( len - bytesRead, maxReadLen );
222  // offset
223  if( offs + bytesRead > m_ptrFileAccessOffset.getMaxValue() )
224  {
225  // we cannot move the inbuffer any further
226  break;
227  }
228  m_ptrFileAccessOffset.write( offs + bytesRead );
229  m_ptrFileAccessLength.write( readSize );
230  // fetch file data into xchange buffer
231  m_ptrFileOperationExecute.call();
232  const std::streamsize result = static_cast<std::streamsize>( m_ptrFileOperationResult.read() );
233  const std::string bufS( m_ptrFileAccessBuffer.readBinary() );
234  memcpy( pBuf + bytesRead, bufS.c_str(), static_cast<size_t>( readSize ) );
235  bytesRead += result;
236  if( m_ptrFileOperationStatus.readS() != "Success" )
237  {
238  // reached end of file
239  break;
240  }
241  }
