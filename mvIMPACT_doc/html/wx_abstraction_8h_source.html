78  static void Save( wxWindow* pWin, const wxString& windowName = wxT( "MainFrame" ) )
79  {
80  wxConfigBase* pConfig( wxConfigBase::Get() );
81  int Height, Width, XPos, YPos;
82  pWin->GetSize( &Width, &Height );
83  pWin->GetPosition( &XPos, &YPos );
84  // when we e.g. try to write config stuff on a read-only file system the result can
85  // be an annoying message box. Therefore we switch off logging during the storage operation.
86  wxLogNull logSuspendScope;
87  pConfig->Write( wxString::Format( wxT( "/%s/h" ), windowName.c_str() ), Height );
88  pConfig->Write( wxString::Format( wxT( "/%s/w" ), windowName.c_str() ), Width );
89  pConfig->Write( wxString::Format( wxT( "/%s/x" ), windowName.c_str() ), XPos );
90  pConfig->Write( wxString::Format( wxT( "/%s/y" ), windowName.c_str() ), YPos );
91  if( dynamic_cast<wxTopLevelWindow*>( pWin ) )
92  {
93  pConfig->Write( wxString::Format( wxT( "/%s/maximized" ), windowName.c_str() ), dynamic_cast<wxTopLevelWindow*>( pWin )->IsMaximized() );
94  }
95  pConfig->Flush();
96  }
97  static wxRect Load( const wxRect& defaultDimensions, bool& boMaximized, const wxString& windowName = wxT( "MainFrame" ) )
98  {
99  wxConfigBase* pConfig( wxConfigBase::Get() );
100  wxRect rect;
101  rect.height = pConfig->Read( wxString::Format( wxT( "/%s/h" ), windowName.c_str() ), defaultDimensions.height );
102  rect.width = pConfig->Read( wxString::Format( wxT( "/%s/w" ), windowName.c_str() ), defaultDimensions.width );
103  rect.x = pConfig->Read( wxString::Format( wxT( "/%s/x" ), windowName.c_str() ), defaultDimensions.x );
104  rect.y = pConfig->Read( wxString::Format( wxT( "/%s/y" ), windowName.c_str() ), defaultDimensions.y );
105  boMaximized = pConfig->Read( wxString::Format( wxT( "/%s/maximized" ), windowName.c_str() ), 1l ) != 0;
106  int displayWidth = 0;
107  int displayHeight = 0;
108  wxDisplaySize( &displayWidth, &displayHeight );
109  if( ( rect.x >= displayWidth ) || ( ( rect.x + rect.width ) < 0 ) )
110  {
111  rect.x = 0;
112  }
113  if( ( rect.y >= displayHeight ) || ( ( rect.y + rect.height ) < 0 ) )
114  {
115  rect.y = 0;
116  }
117  return rect;
118  }
